    ```typescript
    // scripts/generate-test-sprite.ts
    import { createCanvas } from '@napi-rs/canvas';
    import { writeFileSync, mkdirSync } from 'fs';

    // Single 64x64 frame -- solid colored box with a visible border
    const canvas = createCanvas(64, 64);
    const ctx = canvas.getContext('2d');

    // Fill with a distinct color (bright green) so it's visually obvious when rendered
    ctx.fillStyle = '#00FF00';
    ctx.fillRect(0, 0, 64, 64);

    // Add a darker border so the frame boundary is visible
    ctx.strokeStyle = '#008800';
    ctx.lineWidth = 2;
    ctx.strokeRect(1, 1, 62, 62);

    // Add a simple "T" marker in the center to indicate "test"
    ctx.fillStyle = '#008800';
    ctx.font = 'bold 24px sans-serif';
    ctx.textAlign = 'center';
    ctx.textBaseline = 'middle';
    ctx.fillText('T', 32, 32);

    mkdirSync('public/sprites/box', { recursive: true });
    writeFileSync('public/sprites/box/sheet.png', canvas.toBuffer('image/png'));
    console.log('Created public/sprites/box/sheet.png (64x64, 1 frame)');
    ```
